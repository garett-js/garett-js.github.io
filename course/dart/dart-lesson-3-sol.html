Задача 1: Простые типы данных

Dart - Решение

void main() {
  // 1. Добавляем правильные типы данных
  int value1 = 42;
  double value2 = 3.14;
  String value3 = "Учусь программировать";
  String value4 = "2024";
  bool isValue = true;
  int? emptyValue = null;

  // 2. Создаем переменную result и сохраняем результат деления
  double result = value1 / value2;
  print('Результат деления value1 на value2: $result');

  // 3. Отбрасываем дробную часть у value2, перемножаем и выводим
  print('Результат умножения: ${value1 * value2.truncate()}');

  // 4. Делаем буквы в value3 заглавными и выводим
  print('Строка в верхнем регистре: ${value3.toUpperCase()}');

  // 5. Выполняем арифметическую операцию с value4
  print('Результат value4 + 1: ${int.parse(value4) + 1}');
}

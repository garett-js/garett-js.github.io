Задача № 2 — Площадь треугольника

Решение

void main() {
  double base = 5.0;
  double height = 10.0;
  double area = (base * height) / 2;
  print('Площадь треугольника с основанием $base и высотой $height равна $area');
}
Вывод

Площадь треугольника с основанием 5.0 и высотой 10.0 равна 25.0

Задача № 3 — Обмен значений переменных

Решение

void main() {
  int a = 10;
  int b = 20;
  print('До обмена: a = $a, b = $b');
  
  int temp = a;
  a = b;
  b = temp;
  
  print('После обмена: a = $a, b = $b');
}
Вывод

До обмена: a = 10, b = 20
После обмена: a = 20, b = 10

Задача № 4 — Средний балл оценок

Решение

void main() {
  int mathGrade = 5;
  int historyGrade = 4;
  int physicsGrade = 5;
  
  double averageGrade = (mathGrade + historyGrade + physicsGrade) / 3;
  
  print('Средний балл: ${averageGrade.toStringAsFixed(2)}');
}
Вывод

Средний балл: 4.67

Задача № 6 — Обмен значений трех переменных

Решение

void main() {
  // Схема а)
  var a = 1;
  var b = 2;
  var c = 3;
  print('--- Схема а) ---');
  print('До обмена: a = $a, b = $b, c = $c');
  
  // new_a = old_b, new_b = old_c, new_c = old_a
  var temp = a;
  a = b;
  b = c;
  c = temp;
  
  print('После обмена: a = $a, b = $b, c = $c');
  print('');

  // Схема б)
  a = 1;
  b = 2;
  c = 3;
  print('--- Схема б) ---');
  print('До обмена: a = $a, b = $b, c = $c');
  
  // new_a = old_c, new_b = old_a, new_c = old_b
  temp = a;
  a = c;
  c = b;
  b = temp;
  
  print('После обмена: a = $a, b = $b, c = $c');
}
Вывод

--- Схема а) ---
До обмена: a = 1, b = 2, c = 3
После обмена: a = 2, b = 3, c = 1

--- Схема б) ---
До обмена: a = 1, b = 2, c = 3
После обмена: a = 3, b = 1, c = 2

Задача № 7 — Возведение в степень через умножение

Решение

void main() {
  double a = 2.0;
  print('Дано число a = $a\n');

  // а) a⁴ за две операции
  double a2 = a * a;       // 1
  double a4 = a2 * a2;   // 2
  print('a⁴ = $a4 (2 операции)');

  // б) a⁶ за три операции
  // a2 уже есть
  double a3 = a2 * a;    // 2
  double a6 = a3 * a3;   // 3
  print('a⁶ = $a6 (3 операции)');

  // в) a⁷ за четыре операции
  // a4 уже есть
  double a6_v2 = a4 * a2; // 3 (альтернативный способ для a⁶)
  double a7 = a6_v2 * a;  // 4
  print('a⁷ = $a7 (4 операции)');
}
Вывод

Дано число a = 2.0

a⁴ = 16.0 (2 операции)
a⁶ = 64.0 (3 операции)
a⁷ = 128.0 (4 операции)

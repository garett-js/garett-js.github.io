# Задача 5: Характеристики персонажа
BASE_HEALTH = 100
character_name = input("Введите имя персонажа: ")

# Задача 6: Калькулятор лута
SILVER_PER_GOLD = 10
COPPER_PER_SILVER = 100

gold_coins = int(input("Сколько золотых монет найдено? "))
silver_coins = int(input("Сколько серебряных монет найдено? "))
copper_coins = int(input("Сколько медных монет найдено? "))

total_copper = (gold_coins * SILVER_PER_GOLD * COPPER_PER_SILVER) + (silver_coins * COPPER_PER_SILVER) + copper_coins
print(f"Общая стоимость добычи: {total_copper} медных монет.")

print("-" * 20)

# Задача 7: Расчет урона в бою
CRIT_MULTIPLIER = 1.5
hero_damage, monster_armor = 25, 10

print(f"Герой атакует с базовым уроном {hero_damage}.")
print(f"У монстра {monster_armor} единиц брони.")

crit_damage = hero_damage * CRIT_MULTIPLIER
print(f"Критический удар! Урон увеличен в {CRIT_MULTIPLIER} раз и составляет {crit_damage}.")

final_damage = crit_damage - monster_armor
print(f"Монстр получает {final_damage} урона.")
